#날짜와 시간계산

![캡처](https://github.com/locksia/codetree/assets/51018392/ca795ff2-027b-441f-be0c-111777bf7607)
```c++
#include <iostream>
using namespace std;

int main() {
    int a,b,c,d;
    cin >> a >> b >> c >> d;

    int result=0;
    int months = a; //입력받은 시작 날짜
    int days = b;
    while(true){
        if(months == c && days == d){ //입력받은 끝 날짜
            break;
        }

        result++; //총 날짜 계산
        days++;

        if(months == 1 && days == 31){ //각 달의 따른 다른 일수계산
            months++;
            days = 0;
        }
        if(months == 2 && days == 28){
            months++;
            days = 0;
        }
        if(months == 3 && days == 31){
            months++;
            days = 0;
        }
        if(months == 4 && days == 30){
            months++;
            days = 0;
        }
        if(months == 5 && days == 31){
            months++;
            days = 0;
        }
        if(months == 6 && days == 30){
            months++;
            days = 0;
        }
        if(months == 7 && days == 31){
            months++;
            days = 0;
        }
        if(months == 8 && days == 31){
            months++;
            days = 0;
        }
        if(months == 9 && days == 30){
            months++;
            days = 0;
        }
        if(months == 10 && days == 31){
            months++;
            days = 0;
        }
        if(months == 11 && days == 30){
            months++;
            days = 0;
        }
    }

    cout << result+1 << endl;
    return 0;
}
```
