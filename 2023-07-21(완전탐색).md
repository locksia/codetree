# 구간단위로 완전탐색
![캡처1](https://github.com/locksia/codetree/assets/51018392/bf0a60fb-4005-4911-bd06-8f87b5ae52dc)
```
#include <iostream>
using namespace std;

int main() {
    int n,k;
    int x;
    char y;
    int ma = 0;
    int start=0;
    cin >> n >> k;
    int a[10000] = {0};

    for(int i=0; i < n; i++){
        cin >> x >> y;
        if(y == 'G'){
            a[x-1] = 1;
        }
        if(y == 'H'){
            a[x-1] = 2;
        }
    }

    for(int i=0; i < 10000-k; i++){
        int result = 0;
        for(int j=start; j < start + k + 1; j++){
            result += a[j];
        }
        if(result > ma){ma = result;}
        start++;
    }

    cout << ma << endl;
    return 0;
}

```
#include <iostream>
using namespace std;

int main() {
    int n;
    double m;
    cin >> n;
    int cnt = 0;
    double a[n] = {0};

    for (int i = 0; i < n; i++) {
        cin >> m;
        a[i] = m;
    }

    for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
            double sum = 0;
            for (int k = i; k <= j; k++) {
                sum += a[k];
            }
            for (int k = i; k <= j; k++) {
                if (sum / (double)(j - i+1) == a[k]) {
                    cnt++;
                    break;
                }
            }
        }
    }
    cout << cnt << endl;
    return 0;
}
